```python
text = 'текст1'         # строками мы называем всё, что записываем в кавычках
text2 = text + 'текст2' # конкатенация (скалдывание) строк
print(text2)
print(text * 50)        # дублирование строк
# индексация - порядковый номер символа в строке
# счёт идёт от нуля
print(text[0])          # первый элемент
print(text[1])          # второй элемент
print(text[-1])         # последний элемент
print(text[-2])         # предпоследний элемент
print()
print(text2)
print(text2[6:])        # вывод всех символов, начиная с 6 индекса (5-го элемента)
print(text2[:6])        # вывод всех символов до 5 (!) индекса включительно
print(text2[2:7])       # вывод всех символов, начиная с индекса 2 до 6 (включительно)
```
