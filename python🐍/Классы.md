```python
# функции внутрии классов (называются методами)
class Cat:
    def say_meow(self): # self - ключевое слово, которое привязывает функцию к экземпляру
        print('Meow')
    def say(self):
        print('hi')
cat1 = Cat()    # создание экземпляра
cat1.say()      # запуск метода через экземляр
```

```python
class Apple:
    # функция, которая хранит переменные экземпляра
    def __init__(self):
        self.whole_amount = 12          # self привязывает атрибут к экземпляру
    def eat(self, number):
        # number - переменная, которая работает только внутри функции
        self.whole_amount -= number     # обращается к переменной экземпляра через self
        return self.whole_amount
my_apple = Apple()
print(my_apple.whole_amount)            # получение переменной экземпляра
print(my_apple.eat(3))                  # вызов функции экземпляра
```

```python
class Cat:
    def __init__(self, name):       # при создании экземпляра будет принимать имя
        self.name = name            # присваивание локальной переменной к переменной экземпляра
        self.legs = 4
        self.whiskers = True
        self.tail = True
        self.color = None           # пустое значение
cat = Cat('Барсик')                 # указываем имя (без него выдаст ошибку)
cat.color = "рыжий"
print(f"У {cat.name} {cat.color} шерсть")
```
