```python
btn1 = Button(text = 'button 1')
btn2 = Button(text = 'button 2')
btn1.pack(fill='y', side=RIGHT)     # fill размещает объект полностью по горизонтали или вертикали
btn2.pack(fill='y', side=RIGHT)     # side размещает объект справа, слева, сверху или снизу
```

```python
btn1 = Button(text = 'button 1')
btn2 = Button(text = 'button 2')
btn1.grid(column = 1, row = 1)      # делит окно на ячейки,
btn2.grid(column = 2, row = 2)      # column - колонка, row - строка
```

```python
btn1 = Button(text = 'button 1')
btn2 = Button(text = 'button 2')
btn1.place(width = 250, height=55, x = 100, y = 50)     # точные ширина и длина,
btn2.place(width = 250, height=55, x = 100, y = 150)    # точные координаты
```

```python
# запуск функции при нажатии на кнопку
def click():
    btn1.destroy()                          # удаление кнопки
```
