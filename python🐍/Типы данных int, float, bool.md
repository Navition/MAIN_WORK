```python
a = 56      # int - целосчисленные
b = 47.1    # float - десятичные, цифры с точкой
c = True    # правда (True) или ложь (False)
```

```python
# арифметика
print('Сложение:', 5 + 3.1)
print('Вычитание:', 76 - 20)
print('Деление:', 5 / 2)
print('Целочисленное деление:', 5 // 2)
print('Деление по модулю:', 5 % 2)
```

```python
# сравнение
s = 5 > 2       # больше
print(s)        # тип данных bool
print(5 >= 5)   # больше или равно
print(4 <= 5)   # меньше или равно
print(5 == 1)   # равно
print(5 != 1)   # не равно
```

```python
# логические операторы
x = True and False
y = True or False
z = not True
print(x, y, z)
```
