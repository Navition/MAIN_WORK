```python
# создание холста
from tkinter import*
root = Tk()
canvas = Canvas(bg ='orange', width = 750, height =550)
canvas.pack()
```

```python
# изменение параметров холста (разные способы)
canvas['width'] = 500
canvas['height'] = 500
canvas.configure(height=600)
```

```python
# фигуры
canvas.create_line(750, 550, 0, 0)                          # линия
canvas.create_rectangle(250, 100, 345, 150)                 # квадрат
canvas.create_oval(50, 50, 150, 100)                        # овал/круг
canvas.create_polygon(10, 150, 40, 100, 70, 150, 80, 180)   # многоугольник
canvas.create_rectangle(250, 100, 345, 150, fill ='yellow',\ # настройки фигуры
                         outline = 'green', width = 3, activefill = 'white')
```

```python
# создание текста
canvas.create_text(500, 250, text='какой-то текст', \   # координаты левого верхнего угла поля
                   fill="black", font=("Helvetica", 70))
```
