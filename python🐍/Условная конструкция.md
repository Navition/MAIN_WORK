```python
a = 10
if 10 < a < 20:             # начало условной конструкции (записывается всегда)
    print('число находится в диапазон от 10 до 20')
elif a > 30 and a < 50:     # elif используется, если нужно применить несколько условий
    print('число находится в диапазон от 30 до 50')
else:                       # в else записываем случай, когда ни одно из условий неверно
    print('число не находится в указанных диапазонах')
```
